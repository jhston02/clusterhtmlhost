norm_data = np.array(data_Norm['data'])

bin_data = np.array(data_Bin['data'])

TF_data = np.array(data_TF['data'])
data_TFIDF = arff.load(open('TFIDF.arff', 'r'))
TFIDF_data = np.array(data_TFIDF['data'])

from scipy.cluster.hierarchy import dendrogram, linkage, cophenet,inconsistent
from scipy.spatial.distance import pdist

norm_data_z = linkage(norm_data, 'average', 'cosine')
bin_data_z = linkage(bin_data, 'average', 'cosine')
tf_data_z = linkage(TF_data, 'average', 'cosine')
tfidf_data_z = linkage(TFIDF_data, 'average', 'cosine')
